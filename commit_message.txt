```r
# Validate project directory exists and is writable
validate_project_dir <- function(dir_path) {
  if (!dir.exists(dir_path)) {
    stop(paste("Project directory does not exist:", dir_path))
  }
  if (!file.access(dir_path, mode = 2) == 0) {
    warning(paste("Project directory may not be writable:", dir_path))
  }
  return(TRUE)
}

# Check required directories
check_required_dirs <- function(base_dir) {
  required <- c("data", "results")
  for (dir in required) {
    dir_path <- file.path(base_dir, dir)
    if (!dir.exists(dir_path)) {
      dir.create(dir_path, recursive = TRUE)
      message(paste("Created directory:", dir_path))
    }
  }
}
```

Benefits:
- Early error detection before attempting data operations
- Helpful error messages guide user to fix issues
- Automatic creation of required directories
- Validates file system permissions
