Added parameter configuration:
```r
# QC filtering parameters (configurable)
qc_params <- list(
  min_features = 200,     # Minimum genes per spot
  max_features = 6000,    # Maximum genes (doublet threshold)
  max_mt_percent = 20,    # Maximum mitochondrial percentage
  min_counts = 500,       # Minimum UMI counts
  max_counts = 30000      # Maximum UMI counts (outlier threshold)
)

# Function to apply QC filters
apply_qc_filters <- function(seurat_obj, params) {
  original_spots <- ncol(seurat_obj)
  
  seurat_obj <- subset(seurat_obj,
                      subset = nFeature_Spatial > params$min_features &
                              nFeature_Spatial < params$max_features &
                              percent.mt < params$max_mt_percent &
                              nCount_Spatial > params$min_counts &
                              nCount_Spatial < params$max_counts)
  
  filtered_spots <- ncol(seurat_obj)
  message(sprintf("Filtered: %d â†’ %d spots (%.1f%% retained)",
                 original_spots, filtered_spots, 
                 100 * filtered_spots / original_spots))
  
  return(seurat_obj)
}
```
